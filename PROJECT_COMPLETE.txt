  bash setup.sh

  python build.py check-tools

在 WSL 中:
  make run                    # 编译并运行
  make clean all              # 清理并编译
  make debug                  # 调试模式
  bash setup.sh               # 自动设置和运行

在 PowerShell 中:
  .\build.ps1 run             # 编译并运行 (推荐)
  .\build.ps1 build           # 仅编译
  .\build.ps1 debug           # 调试模式
  .\build.ps1 clean           # 清理
  .\build.ps1 wsl             # 打开 WSL

Python:
  python build.py run         # 编译并运行
  python build.py check-tools # 检查工具
